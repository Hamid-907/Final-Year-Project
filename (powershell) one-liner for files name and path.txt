$outputFile="$env:USERPROFILE\Desktop\document_file_paths.txt"; (Get-PSDrive -PSProvider FileSystem | ForEach-Object { Get-ChildItem -Path $_.Root -Recurse -File -ErrorAction SilentlyContinue -Include *.pdf,*.docx,*.xlsx }) + (Get-ChildItem -Path "$env:USERPROFILE" -Recurse -File -ErrorAction SilentlyContinue -Include *.pdf,*.docx,*.xlsx) | Sort-Object LastWriteTime -Descending | ForEach-Object { "$($_.LastWriteTime) - $($_.FullName)" } | Out-File -FilePath $outputFile -Encoding utf8; Write-Host "Document file paths have been saved to $outputFile, sorted by last modified date."
